\documentclass{standalone}
\RequirePackage{tikz}
\RequirePackage{ifthen}
\usetikzlibrary{arrows,shapes,matrix}
\tikzstyle{btreeptr} = [draw, semithick, fill=blue!50, minimum height=1cm, minimum width=0.5cm]
\tikzstyle{btreeval} = [draw, semithick, fill=yellow!30, minimum size=1cm]
\tikzstyle{btreevale} = [draw,semithick, fill=green!30, minimum size=1cm]
\tikzstyle{btlink} = [draw, semithick, ->, >=triangle 45]
\newcommand{\xyshift}[3]{
  \begin{scope}[xshift=#1, yshift=#2]
    #3
  \end{scope}
}
\newcommand{\btreelink}[2]{
\draw[btlink] (#1.south) -- ([yshift=-0.1cm] #2.north); \&
\fill ([yshift=0.5cm]  #1.south) circle[radius=0.15cm]; \\
}
\newcommand{\btreeinodea}[2]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreeptr] (#1-a) {\vphantom{1}}; \& \node[btreeval] {#2}; \&
 \node[btreeptr] (#1-b) {\vphantom{1}}; \\
};
}
\newcommand{\btreelnodea}[2]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreevale] {#2}; \\
};
}
\newcommand{\btreeinodeb}[3]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreeptr] (#1-a) {\vphantom{1}}; \& \node[btreeval] {#2}; \&
 \node[btreeptr] (#1-b) {\vphantom{1}}; \& \node[btreeval] {#3}; \&
 \node[btreeptr] (#1-c) {\vphantom{1}}; \\
};
}
\newcommand{\btreelnodeb}[3]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreevale] {#2}; \&
 \node[btreevale] {#3}; \\
};
}
\newcommand{\btreeinodec}[4]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreeptr] (#1-a) {\vphantom{1}}; \& \node[btreeval] {#2}; \&
 \node[btreeptr] (#1-b) {\vphantom{1}}; \& \node[btreeval] {#3}; \&
 \node[btreeptr] (#1-c) {\vphantom{1}}; \& \node[btreeval] {#4}; \&
 \node[btreeptr] (#1-d) {\vphantom{1}}; \\
};
}
\newcommand{\btreelnodec}[4]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreevale] {#2}; \&
 \node[btreevale] {#3}; \&
 \node[btreevale] {#4}; \\
};
}
\newcommand{\btreeinoded}[5]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreeptr] (#1-a) {\vphantom{1}}; \& \node[btreeval] {#2}; \&
 \node[btreeptr] (#1-b) {\vphantom{1}}; \& \node[btreeval] {#3}; \&
 \node[btreeptr] (#1-c) {\vphantom{1}}; \& \node[btreeval] {#4}; \&
 \node[btreeptr] (#1-d) {\vphantom{1}}; \& \node[btreeval] {#5}; \&
 \node[btreeptr] (#1-e) {\vphantom{1}}; \\
};
}
\newcommand{\btreelnoded}[5]{
\matrix [ampersand replacement=\&] (#1)
{
 \node[btreevale] {#2}; \&
 \node[btreevale] {#3}; \&
 \node[btreevale] {#4}; \&
 \node[btreevale] {#5}; \\
};
}

\begin{document}
\begin{tikzpicture}
\tikzstyle{every node}=[font=\fontsize{15}{0}\selectfont]
\xyshift{25mm}{0mm}{\btreelnoded{a}{a}{b}{c}{d}}
\xyshift{70mm}{0mm}{\btreelnodec{b}{f}{g}{h}}
\xyshift{105mm}{0mm}{\btreelnodeb{c}{j}{k}}
\xyshift{135mm}{0mm}{\btreelnodeb{d}{o}{p}}
\xyshift{80mm}{30mm}{\btreeinodec{e}{e}{i}{l}}
\btreelink{e-a}{a}
\btreelink{e-b}{b}
\btreelink{e-c}{c}
\btreelink{e-d}{d}
\end{tikzpicture}
\end{document}
